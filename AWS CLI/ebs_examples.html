Example 4: Complete and Tag Snapshot
#!/bin/bash
# Complete snapshot and apply metadata tags
SNAP_ID="snap-0abc123def456789"
aws ebs complete-snapshot \
  --snapshot-id $SNAP_ID \
  --changed-blocks-count 150

# Tag completed snapshot
aws ec2 create-tags --resources $SNAP_ID --tags \
  Key=Environment,Value=production \
  Key=Application,Value=anihpj \
  Key=BackupType,Value=incremental
{ "Status": "completed" }
# Tagged backup from jobpost.models import Skills, User, Author, Location, JobPost print(f"Snapshot completed and tagged") print(f"Environment: production") print(f"Application: anihpj") print(f"Type: incremental") # Log tagged backup backup_meta = { 'snapshot_id': 'snap-0abc123def456789', 'environment': 'production', 'app': 'anihpj', 'type': 'incremental' } print(f"Metadata: {backup_meta}")
Example 5: Complete with Monitoring
#!/bin/bash
# Complete snapshot with CloudWatch monitoring
SNAP_ID="snap-0abc123def456789"
TIMESTAMP=$(date '+%Y-%m-%d %H:%M:%S')

echo "Completing snapshot at $TIMESTAMP"
aws ebs complete-snapshot \
  --snapshot-id $SNAP_ID \
  --changed-blocks-count 150

# Send completion metric to CloudWatch
aws cloudwatch put-metric-data \
  --namespace "EBS/Snapshots" \
  --metric-data MetricName=CompletedSnapshots,Value=1
