sudo mkdir -p /opt/anihpj

# Create virtual environment
python3 -m venv venv

pip install -r requirements.txt

# Install additional packages for production
pip install gunicorn psycopg2-binary

cat > .env << EOL
DEBUG=False
SECRET_KEY='your-secret-key-here'
DATABASE_URL=postgresql://username:password@anihpj-db.region.rds.amazonaws.com:5432/anihpj_db

[Install]

        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;

sudo ln -s /etc/nginx/sites-available/anihpj /etc/nginx/sites-enabled/

Optimized Django Views for Your Models
⚡ Optimized JobPost Views
# jobpost/views.py - Optimized for EC2 deployment
from django.shortcuts import render, get_object_or_404
from django.core.paginator import Paginator
from django.db.models import Prefetch, Count, Q
from django.views.generic import ListView
from django.http import JsonResponse
from django.core.cache import cache
from .models import JobPost, Skills, Author, Location

class OptimizedJobListView(ListView):
    """Optimized job listing view for EC2 deployment"""
    model = JobPost
    template_name = 'jobpost/job_list.html'
    context_object_name = 'job_posts'
    paginate_by = 20

    def get_queryset(self):
        # Use select_related for foreign keys to reduce database queries
        queryset = JobPost.objects.select_related(
            'author__user',  # Get user info with author
            'location'       # Get location info
        ).prefetch_related(
            'skills'         # Prefetch many-to-many skills
        ).order_by('-date')
        
        # Add filtering
        location_filter = self.request.GET.get('location')
        skill_filter = self.request.GET.get('skill')
        job_type = self.request.GET.get('type')
        
        if location_filter:
            queryset = queryset.filter(
                Q(location__city__icontains=location_filter) |
                Q(location__state__icontains=location_filter) |
                Q(location__country__icontains=location_filter)
            )
            
        if skill_filter:
            queryset = queryset.filter(skills__name__icontains=skill_filter)
            
        if job_type:
            queryset = queryset.filter(type=job_type)
            
        return queryset.distinct()

    def get_context_data(self, **kwargs):
        context = super().get_context_data(**kwargs)
        
        # Cache expensive queries for 5 minutes
        skills_cache_key = 'all_skills_list'
        skills = cache.get(skills_cache_key)
        if skills is None:
            skills = Skills.objects.all().order_by('name')
            cache.set(skills_cache_key, skills, 300)  # 5 minutes
        
        locations_cache_key = 'unique_locations'
        locations = cache.get(locations_cache_key)
        if locations is None:
            locations = Location.objects.values('city', 'state', 'country').distinct()
            cache.set(locations_cache_key, locations, 300)
            
        context.update({
            'skills': skills,
            'locations': locations,
            'job_types': JobPost.JOB_TYPE_CHOICES,
        })
        return context

def job_detail_api(request, slug):
    """API endpoint for job details - cached for performance"""
    cache_key = f'job_detail_{slug}'
    job_data = cache.get(cache_key)
    
    if job_data is None:
        try:
            job = JobPost.objects.select_related(
                'author__user', 'location'
            ).prefetch_related('skills').get(slug=slug)
            
            job_data = {
                'id': job.id,
                'title': job.title,
                'description': job.description,
                'salary': job.salary,
                'type': job.type,
                'date': job.date.isoformat(),
                'expiry': job.expiry.isoformat() if job.expiry else None,
                'author': {
                    'name': f"{job.author.user.first_name} {job.author.user.last_name}",
                    'company': job.author.company,
                    'designation': job.author.designation,
                },
                'location': {
                    'street': job.location.street,
                    'city': job.location.city,
                    'state': job.location.state,
                    'country': job.location.country,
                    'zip': job.location.zip,
                },
                'skills': [skill.name for skill in job.skills.all()]
            }
            cache.set(cache_key, job_data, 600)  # 10 minutes
            
        except JobPost.DoesNotExist:
            return JsonResponse({'error': 'Job not found'}, status=404)
    
    return JsonResponse(job_data)

def analytics_dashboard(request):
    """Analytics dashboard with cached statistics"""
    cache_key = 'job_analytics_data'
    analytics = cache.get(cache_key)
    
    if analytics is None:
        analytics = {
            'total_jobs': JobPost.objects.count(),
            'active_jobs': JobPost.objects.filter(expiry__gte=timezone.now().date()).count(),
            'jobs_by_type': dict(JobPost.objects.values_list('type').annotate(count=Count('type'))),

    return render(request, 'jobpost/analytics.html', {'analytics': analytics})

7. Load Testing and Performance

🔧 Load Testing Script for Job Portal
# load_test.py - Test your Django app on EC2
import requests
import time
import concurrent.futures
from datetime import datetime

BASE_URL = 'http://your-ec2-instance.amazonaws.com'

def test_job_list():
    """Test job listing endpoint"""
    try:
        response = requests.get(f'{BASE_URL}/jobs/')
        return response.status_code, response.elapsed.total_seconds()
    except Exception as e:
        return 0, 0

def test_job_detail():
    """Test job detail endpoint"""
    try:
        response = requests.get(f'{BASE_URL}/jobs/python-developer/')
        return response.status_code, response.elapsed.total_seconds()
    except Exception as e:
        return 0, 0

def test_api_endpoint():
    """Test API endpoint"""
    try:
        response = requests.get(f'{BASE_URL}/api/jobs/python-developer/')
        return response.status_code, response.elapsed.total_seconds()
    except Exception as e:
        return 0, 0

def run_load_test(num_requests=100, concurrent_users=10):
    """Run load test with multiple concurrent users"""
    
    test_functions = [test_job_list, test_job_detail, test_api_endpoint]
    results = []
    
    print(f"Starting load test: {num_requests} requests with {concurrent_users} concurrent users")
    print(f"Testing at: {datetime.now()}")
    
    start_time = time.time()
    
    with concurrent.futures.ThreadPoolExecutor(max_workers=concurrent_users) as executor:
        # Submit all test requests
        futures = []
        for i in range(num_requests):
            test_func = test_functions[i % len(test_functions)]
            future = executor.submit(test_func)
            futures.append(future)
        
        # Collect results
        for future in concurrent.futures.as_completed(futures):
            status_code, response_time = future.result()
            results.append((status_code, response_time))
    
    end_time = time.time()
    
    # Analyze results
    successful_requests = [r for r in results if r[0] == 200]
    response_times = [r[1] for r in successful_requests]
    
    if response_times:
        avg_response_time = sum(response_times) / len(response_times)
        max_response_time = max(response_times)
        min_response_time = min(response_times)
    else:
        avg_response_time = max_response_time = min_response_time = 0
    
    total_time = end_time - start_time
    success_rate = len(successful_requests) / len(results) * 100
    
    print("\n" + "="*50)
    print("LOAD TEST RESULTS")
    print("="*50)
    print(f"Total requests: {num_requests}")
    print(f"Successful requests: {len(successful_requests)}")
    print(f"Success rate: {success_rate:.2f}%")
    print(f"Total test time: {total_time:.2f} seconds")
    print(f"Requests per second: {num_requests/total_time:.2f}")
    print(f"Average response time: {avg_response_time:.3f} seconds")
    print(f"Max response time: {max_response_time:.3f} seconds")
    print(f"Min response time: {min_response_time:.3f} seconds")

if __name__ == "__main__":
    run_load_test(200, 20)  # 200 requests with 20 concurrent users
