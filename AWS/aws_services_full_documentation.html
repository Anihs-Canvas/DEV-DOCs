Amazon EC2 (Elastic Compute Cloud)

Category: Compute  |  Use Case: Web Application Hosting  |  Pricing: On-Demand, Reserved, Spot Instances  |  Last Updated: November 2025  |  Documentation: AWS EC2 Docs

Secure and resizable compute capacity in the cloud for hosting Django applications with automatic scaling, load balancing, and high availability.

📋 Overview & Introduction

Amazon Elastic Compute Cloud (EC2) is a web service that provides secure, resizable compute capacity in the cloud. It's designed to make web-scale cloud computing easier for developers by providing complete control over computing resources.

🎯 Key Benefits
  • Elastic Computing: Scale capacity up or down within minutes
  • Complete Control: Root access to instances and full control over computing resources
  • Flexible Pricing: Pay only for what you use with multiple pricing models
  • Integrated: Seamlessly works with other AWS services
  • Secure: Built-in firewall, VPC, and IAM integration
  • Reliable: 99.99% SLA with multiple availability zones
💡 When to Use EC2: Perfect for your anihpj Django job portal when you need full control over the server environment, custom configurations, or need to run background tasks, scheduled jobs, or WebSocket connections alongside your web application.
⚖️ EC2 vs. Other Compute Services
EC2: Full control, persistent, traditional apps
Lambda: Serverless, event-driven, short tasks
ECS/EKS: Container orchestration
Elastic Beanstalk: Managed platform for web apps
Lightsail: Simplified VPS for simple apps
Fargate: Serverless containers

🔍 Instance Naming Convention Explained
Example: m5.xlarge

┌─┬──┬──────┐
│m│5 │xlarge│
└┬┘└┬┘└──┬───┘
 │  │   └─→ Size: nano, micro, small, medium, large, xlarge, 2xlarge, etc.
 │  │       • xlarge = 4 vCPUs, 16 GiB RAM (for m5 family)
 │  │       • 2xlarge = 8 vCPUs, 32 GiB RAM
 │  │
 │  └─→ Generation: 5 (latest generation for m family)
 │       • Higher generation = better performance & newer features
 │       • Always prefer latest generation when available
 │
 └─→ Family: m (General Purpose, balanced compute/memory)
      • t = Burstable
      • m = General Purpose
      • c = Compute Optimized
      • r = Memory Optimized
      • i = Storage Optimized

# anihpj/middleware/cloudwatch_metrics.py
import boto3
from django.utils.deprecation import MiddlewareMixin
from datetime import datetime

class CloudWatchMetricsMiddleware(MiddlewareMixin):
    """Send custom Django metrics to CloudWatch"""
    
    def __init__(self, get_response):
        super().__init__(get_response)
        self.cloudwatch = boto3.client('cloudwatch', region_name='us-east-1')
    
    def process_response(self, request, response):
        # Track response times
        if hasattr(request, '_start_time'):
            response_time = (datetime.now() - request._start_time).total_seconds()
            
            self.cloudwatch.put_metric_data(
                Namespace='AnihpjJobPortal',
                MetricData=[
                    {
                        'MetricName': 'ResponseTime',
                        'Value': response_time,
                        'Unit': 'Seconds',
                        'Dimensions': [
                            {'Name': 'View', 'Value': request.path},
                            {'Name': 'Method', 'Value': request.method}
                        ]
                    }
                ]
            )
        
        # Track HTTP status codes
        self.cloudwatch.put_metric_data(
            Namespace='AnihpjJobPortal',
            MetricData=[
                {
                    'MetricName': 'HTTPStatusCode',
                    'Value': 1,
                    'Unit': 'Count',
                    'Dimensions': [
                        {'Name': 'StatusCode', 'Value': str(response.status_code)},
                        {'Name': 'Path', 'Value': request.path}
                    ]
                }
            ]
        )
        
        return response
    
    def process_request(self, request):
        request._start_time = datetime.now()

# anihpj/jobpost/views.py
from django.http import JsonResponse
from django.db import connection
from django.core.cache import cache
import logging

logger = logging.getLogger(__name__)

def health_check(request):
    """
    Health check endpoint for ALB
    Returns 200 if healthy, 503 if unhealthy
    """
    health_status = {
        'status': 'healthy',
        'checks': {}
    }
    is_healthy = True
    
    # Check database connectivity
    try:
        with connection.cursor() as cursor:
            cursor.execute("SELECT 1")
            health_status['checks']['database'] = 'ok'
    except Exception as e:
        health_status['checks']['database'] = f'error: {str(e)}'
        is_healthy = False
        logger.error(f"Database health check failed: {e}")
    
    # Check cache connectivity
    try:
        cache.set('health_check', 'ok', 10)
        if cache.get('health_check') == 'ok':
            health_status['checks']['cache'] = 'ok'
        else:
            health_status['checks']['cache'] = 'error'
            is_healthy = False
    except Exception as e:
        health_status['checks']['cache'] = f'error: {str(e)}'
        is_healthy = False
        logger.error(f"Cache health check failed: {e}")
    
    # Check disk space
    import shutil
    try:
        stat = shutil.disk_usage('/')
        free_percent = (stat.free / stat.total) * 100
        if free_percent < 10:
            health_status['checks']['disk'] = f'warning: {free_percent:.1f}% free'
            is_healthy = False
        else:
            health_status['checks']['disk'] = f'ok: {free_percent:.1f}% free'
    except Exception as e:
        health_status['checks']['disk'] = f'error: {str(e)}'
        is_healthy = False
    
    if is_healthy:
        return JsonResponse(health_status, status=200)
    else:
        health_status['status'] = 'unhealthy'
        return JsonResponse(health_status, status=503)

import json
from datetime import datetime

def process_resume_batch(s3_bucket, s3_keys):
    """
    Process multiple resumes in batch
    Runs on EC2 Spot instances via AWS Batch
    """
    s3 = boto3.client('s3')
    results = []
    
    for key in s3_keys:
        try:
            # Download resume from S3
            response = s3.get_object(Bucket=s3_bucket, Key=key)
            pdf_content = response['Body'].read()
            
            # Parse PDF
            pdf_reader = PyPDF2.PdfReader(io.BytesIO(pdf_content))
            text = ''
            for page in pdf_reader.pages:
                text += page.extract_text()
            
            # Extract structured data (simplified)
            parsed_data = {
                'filename': key,
                'text': text,
                'email': extract_email(text),
                'phone': extract_phone(text),
                'skills': extract_skills(text),
                'experience_years': extract_experience(text),
                'processed_at': datetime.utcnow().isoformat()
            }
            
            # Store results in S3 or database
            result_key = key.replace('uploads/', 'parsed/').replace('.pdf', '.json')
            s3.put_object(
                Bucket=s3_bucket,
                Key=result_key,
                Body=json.dumps(parsed_data),
                ContentType='application/json'
            )
            
            results.append({'key': key, 'status': 'success'})
            
        except Exception as e:
            results.append({'key': key, 'status': 'error', 'error': str(e)})
    
    return results

💰 Common FAQs
Q: Why am I being charged when my instance is stopped?
A: Stopped instances don't incur EC2 compute charges, but you're 
   still charged for:
   • EBS volumes attached to the instance
   • Elastic IP addresses not associated with running instances
   • Snapshots and AMIs
   
   Solution: Delete unused EBS volumes, release unused Elastic IPs

Q: What's the difference between Stop and Terminate?
A: • Stop: Instance is stopped, can be started again, data preserved
   • Terminate: Instance is permanently deleted, cannot be recovered
   
   Tip: Enable termination protection for production instances!

Q: Can I change instance type without losing data?
A: Yes! Stop the instance, change instance type, then start it.
   Data on EBS volumes is preserved.
   
   aws ec2 stop-instances --instance-ids i-0abc123
   aws ec2 modify-instance-attribute \
       --instance-id i-0abc123 \
       --instance-type m5.xlarge
   aws ec2 start-instances --instance-ids i-0abc123

Q: How do I transfer files to/from EC2?
A: Multiple options:
   • SCP: scp -i key.pem file.txt ec2-user@1.2.3.4:/home/ec2-user/
   • SFTP: sftp -i key.pem ec2-user@1.2.3.4
   • S3: Upload to S3, then download on EC2 (aws s3 cp)
   • rsync: rsync -avz -e "ssh -i key.pem" ...

Q: What happens to my data when instance is terminated?
A: • Root EBS volume: Deleted (unless DeleteOnTermination=false)
   • Additional EBS volumes: Preserved (unless explicitly deleted)
   • Instance Store: Permanently lost
   
   Always backup important data!

Q: Can I recover a terminated instance?
A: No, termination is permanent. However:
   • If you have EBS snapshots, create new instance from snapshot
   • If you have AMI backup, launch new instance from AMI
   • Check CloudTrail to see who terminated it and when

Q: How do I get a static IP for my EC2 instance?
A: Allocate and associate an Elastic IP:
   
   aws ec2 allocate-address --domain vpc
   aws ec2 associate-address \
       --instance-id i-0abc123 \
       --allocation-id eipalloc-0abc123

Q: Can I use EC2 without a VPC?
A: No, EC2-Classic has been retired. All EC2 instances must be 
   launched in a VPC (Virtual Private Cloud).

# anihpj/management/commands/manage_ec2.py
import boto3
from django.core.management.base import BaseCommand

class Command(BaseCommand):
    help = 'Manage EC2 instances for anihpj'
    
    def __init__(self):
        super().__init__()
        self.ec2 = boto3.client('ec2', region_name='us-east-1')
        self.ec2_resource = boto3.resource('ec2', region_name='us-east-1')
    
    def add_arguments(self, parser):
        parser.add_argument('action', type=str, 
                          help='Action: list|start|stop|launch')
        parser.add_argument('--instance-id', type=str, 
                          help='Instance ID for start/stop actions')
    
    def handle(self, *args, **options):
        action = options['action']
        
        if action == 'list':
            self.list_instances()
        elif action == 'start':
            self.start_instance(options['instance_id'])
        elif action == 'stop':
            self.stop_instance(options['instance_id'])
        elif action == 'launch':
            self.launch_instance()
    
    def list_instances(self):
        """List all anihpj instances"""
        response = self.ec2.describe_instances(
            Filters=[
                {'Name': 'tag:Project', 'Values': ['anihpj']},
                {'Name': 'instance-state-name', 
                 'Values': ['running', 'stopped', 'stopping', 'pending']}
            ]
        )
        
        for reservation in response['Reservations']:
            for instance in reservation['Instances']:
                name = next((tag['Value'] for tag in instance.get('Tags', []) 
                           if tag['Key'] == 'Name'), 'N/A')
                
                self.stdout.write(
                    f"ID: {instance['InstanceId']}\n"
                    f"Name: {name}\n"
                    f"Type: {instance['InstanceType']}\n"
                    f"State: {instance['State']['Name']}\n"
                    f"IP: {instance.get('PublicIpAddress', 'N/A')}\n"
                    f"Launch Time: {instance['LaunchTime']}\n"
                    f"---"
                )
    
    def start_instance(self, instance_id):
        """Start stopped instance"""
        if not instance_id:
            self.stderr.write("Error: --instance-id required")
            return
        
        try:
            self.ec2.start_instances(InstanceIds=[instance_id])
            self.stdout.write(
                self.style.SUCCESS(f"Started instance {instance_id}")
            )
        except Exception as e:
            self.stderr.write(f"Error starting instance: {str(e)}")
    
    def stop_instance(self, instance_id):
        """Stop running instance"""
        if not instance_id:
            self.stderr.write("Error: --instance-id required")
            return
        
        try:
            self.ec2.stop_instances(InstanceIds=[instance_id])
            self.stdout.write(
                self.style.SUCCESS(f"Stopped instance {instance_id}")
            )
        except Exception as e:
            self.stderr.write(f"Error stopping instance: {str(e)}")
    
    def launch_instance(self):
        """Launch new anihpj web server"""
        try:
            instances = self.ec2_resource.create_instances(
                ImageId='ami-0c7217cdde317cfec',  # Amazon Linux 2023
                InstanceType='t3.medium',
                KeyName='anihpj-keypair',
                SecurityGroupIds=['sg-0abc123'],
                SubnetId='subnet-0abc123',
                MinCount=1,
                MaxCount=1,
                TagSpecifications=[
                    {
                        'ResourceType': 'instance',
                        'Tags': [
                            {'Key': 'Name', 'Value': 'anihpj-web-auto'},
                            {'Key': 'Project', 'Value': 'anihpj'},
                            {'Key': 'Environment', 'Value': 'production'},
                            {'Key': 'ManagedBy', 'Value': 'django-command'}
                        ]
                    }
                ],
                IamInstanceProfile={'Name': 'anihpj-ec2-profile'},
                UserData=self.get_user_data()
            )
            
            instance_id = instances[0].id

            instances[0].reload()
            
            self.stdout.write(
                self.style.SUCCESS(

                )

📖 Glossary

Key EC2 Terms:

AMI (Amazon Machine Image)
    Pre-configured template for EC2 instances containing OS and software.

Auto Scaling
    Automatically adjust the number of EC2 instances based on demand.

Availability Zone (AZ)
    Isolated data center within an AWS Region. Each Region has multiple AZs.

EBS (Elastic Block Store)
    Persistent block storage for EC2 instances (like a hard drive).

EC2 (Elastic Compute Cloud)
    Virtual servers in the AWS cloud.

Elastic IP
    Static IPv4 address designed for dynamic cloud computing.

IAM (Identity and Access Management)
    Service for managing access to AWS resources.

Instance
    A virtual server in Amazon EC2.

Instance Store
    Temporary block storage physically attached to host computer.

Launch Template
    Configuration template for launching EC2 instances.

Reserved Instance
    Purchase option with discounted hourly rate for 1 or 3-year commitment.

Security Group
    Virtual firewall controlling inbound and outbound traffic to instances.

Snapshot
    Point-in-time backup of an EBS volume.

Spot Instance
    Spare EC2 capacity available at up to 90% discount.

User Data
    Scripts or commands that run when an instance starts.

VPC (Virtual Private Cloud)
    Isolated virtual network where you launch AWS resources.

class OptimizedJobListView(ListView):
    """Optimized job listing view for EC2 deployment"""
    model = JobPost
    template_name = 'jobpost/job_list.html'
    context_object_name = 'job_posts'
    paginate_by = 20

    def get_queryset(self):
        # Use select_related for foreign keys to reduce database queries
        queryset = JobPost.objects.select_related(
            'author__user',  # Get user info with author
            'location'       # Get location info
        ).prefetch_related(
            'skills'         # Prefetch many-to-many skills
        ).order_by('-date')
        
        # Add filtering
        location_filter = self.request.GET.get('location')
        skill_filter = self.request.GET.get('skill')
        job_type = self.request.GET.get('type')
        
        if location_filter:
            queryset = queryset.filter(
                Q(location__city__icontains=location_filter) |
                Q(location__state__icontains=location_filter) |
                Q(location__country__icontains=location_filter)
            )
            
        if skill_filter:
            queryset = queryset.filter(skills__name__icontains=skill_filter)
            
        if job_type:
            queryset = queryset.filter(type=job_type)
            
        return queryset.distinct()

    def get_context_data(self, **kwargs):
        context = super().get_context_data(**kwargs)
        
        # Cache expensive queries for 5 minutes
        skills_cache_key = 'all_skills_list'
        skills = cache.get(skills_cache_key)
        if skills is None:
            skills = Skills.objects.all().order_by('name')
            cache.set(skills_cache_key, skills, 300)  # 5 minutes
        
        locations_cache_key = 'unique_locations'
        locations = cache.get(locations_cache_key)
        if locations is None:
            locations = Location.objects.values('city', 'state', 'country').distinct()
            cache.set(locations_cache_key, locations, 300)
            
        context.update({
            'skills': skills,
            'locations': locations,
            'job_types': JobPost.JOB_TYPE_CHOICES,
        })
        return context

import requests

    except Exception as e:

    except Exception as e:

def run_load_test(num_requests=100, concurrent_users=10):
    """Run load test with multiple concurrent users"""
    
    test_functions = [test_job_list, test_job_detail, test_api_endpoint]
    results = []
    
    print(f"Starting load test: {num_requests} requests with {concurrent_users} concurrent users")
    print(f"Testing at: {datetime.now()}")
    
    start_time = time.time()
    
    with concurrent.futures.ThreadPoolExecutor(max_workers=concurrent_users) as executor:
        # Submit all test requests
        futures = []
        for i in range(num_requests):
            test_func = test_functions[i % len(test_functions)]
            future = executor.submit(test_func)
            futures.append(future)
        
        # Collect results
        for future in concurrent.futures.as_completed(futures):
            status_code, response_time = future.result()
            results.append((status_code, response_time))
    
    end_time = time.time()
    
    # Analyze results
    successful_requests = [r for r in results if r[0] == 200]
    response_times = [r[1] for r in successful_requests]
    
    if response_times:
        avg_response_time = sum(response_times) / len(response_times)
        max_response_time = max(response_times)
        min_response_time = min(response_times)
    else:
        avg_response_time = max_response_time = min_response_time = 0
    
    total_time = end_time - start_time
    success_rate = len(successful_requests) / len(results) * 100
    
    print("\n" + "="*50)
    print("LOAD TEST RESULTS")
    print("="*50)
    print(f"Total requests: {num_requests}")
    print(f"Successful requests: {len(successful_requests)}")

    print(f"Total test time: {total_time:.2f} seconds")
