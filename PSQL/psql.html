Table Definitions

📁 departments

Stores company department information

CREATE TABLE departments (
    dept_id       SERIAL PRIMARY KEY,
    dept_name     VARCHAR(100) NOT NULL UNIQUE,
    location      VARCHAR(100),
    budget        DECIMAL(15,2) DEFAULT 0.00,
    created_at    TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

👤 employees

Employee records with self-referencing manager relationship

CREATE TABLE employees (
    emp_id        SERIAL PRIMARY KEY,
    first_name    VARCHAR(50) NOT NULL,
    last_name     VARCHAR(50) NOT NULL,
    email         VARCHAR(100) UNIQUE NOT NULL,
    phone         VARCHAR(20),
    hire_date     DATE NOT NULL DEFAULT CURRENT_DATE,
    salary        DECIMAL(10,2) CHECK (salary > 0),
    commission    DECIMAL(5,2) DEFAULT 0.00,
    dept_id       INTEGER REFERENCES departments(dept_id),
    manager_id    INTEGER REFERENCES employees(emp_id),
    status        VARCHAR(20) DEFAULT 'active' 
                  CHECK (status IN ('active', 'inactive', 'terminated')),
    metadata      JSONB DEFAULT '{}'
);

🎯 skills

Technical and soft skills catalog

CREATE TABLE skills (
    skill_id      SERIAL PRIMARY KEY,
    skill_name    VARCHAR(100) NOT NULL UNIQUE,
    category      VARCHAR(50) NOT NULL
                  CHECK (category IN ('technical', 'soft', 'language', 'certification'))
);

🔗 employee_skills (Junction Table)

Many-to-many relationship between employees and skills

CREATE TABLE employee_skills (
    emp_id        INTEGER REFERENCES employees(emp_id) ON DELETE CASCADE,
    skill_id      INTEGER REFERENCES skills(skill_id) ON DELETE CASCADE,
    proficiency   INTEGER CHECK (proficiency BETWEEN 1 AND 5),
    certified     BOOLEAN DEFAULT FALSE,
    acquired_date DATE DEFAULT CURRENT_DATE,
    PRIMARY KEY (emp_id, skill_id)
);

🛒 customers

Customer information with JSON address storage

CREATE TABLE customers (
    customer_id    SERIAL PRIMARY KEY,
    first_name     VARCHAR(50) NOT NULL,
    last_name      VARCHAR(50) NOT NULL,
    email          VARCHAR(100) UNIQUE NOT NULL,
    phone          VARCHAR(20),
    address        JSONB DEFAULT '{}',
    created_at     TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    loyalty_points INTEGER DEFAULT 0 CHECK (loyalty_points >= 0)
);

📂 categories

Hierarchical product categories (self-referencing)

CREATE TABLE categories (
    category_id   SERIAL PRIMARY KEY,
    category_name VARCHAR(100) NOT NULL,
    parent_id     INTEGER REFERENCES categories(category_id),
    description   TEXT
);

🏭 suppliers

Product supplier/vendor information

CREATE TABLE suppliers (
    supplier_id   SERIAL PRIMARY KEY,
    company_name  VARCHAR(150) NOT NULL,
    contact_name  VARCHAR(100),
    contact_email VARCHAR(100),
    phone         VARCHAR(20),
    address       VARCHAR(200),
    city          VARCHAR(100),
    country       VARCHAR(100) DEFAULT 'USA'
);

📦 products

Product catalog with arrays and JSON specifications

CREATE TABLE products (
    product_id     SERIAL PRIMARY KEY,
    product_name   VARCHAR(200) NOT NULL,
    category_id    INTEGER REFERENCES categories(category_id),
    supplier_id    INTEGER REFERENCES suppliers(supplier_id),
    unit_price     DECIMAL(10,2) NOT NULL CHECK (unit_price >= 0),
    units_in_stock INTEGER DEFAULT 0 CHECK (units_in_stock >= 0),
    units_on_order INTEGER DEFAULT 0,
    discontinued   BOOLEAN DEFAULT FALSE,
    tags           TEXT[] DEFAULT '{}',
    specs          JSONB DEFAULT '{}'
);

📋 orders

Customer orders with status tracking

CREATE TABLE orders (
    order_id       SERIAL PRIMARY KEY,
    customer_id    INTEGER NOT NULL REFERENCES customers(customer_id),
    employee_id    INTEGER REFERENCES employees(emp_id),
    order_date     TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    shipped_date   TIMESTAMP,
    status         VARCHAR(20) DEFAULT 'pending'
                   CHECK (status IN ('pending', 'processing', 'shipped', 'delivered', 'cancelled')),
    total_amount   DECIMAL(12,2) DEFAULT 0.00,
    notes          TEXT
);

📝 order_items

Line items for each order

CREATE TABLE order_items (
    order_id       INTEGER REFERENCES orders(order_id) ON DELETE CASCADE,
    product_id     INTEGER REFERENCES products(product_id),
    quantity       INTEGER NOT NULL CHECK (quantity > 0),
    unit_price     DECIMAL(10,2) NOT NULL,
    discount       DECIMAL(5,2) DEFAULT 0.00 CHECK (discount >= 0 AND discount <= 100),
    PRIMARY KEY (order_id, product_id)
);

⭐ reviews

Product reviews and ratings

CREATE TABLE reviews (
    review_id      SERIAL PRIMARY KEY,
    product_id     INTEGER NOT NULL REFERENCES products(product_id),
    customer_id    INTEGER NOT NULL REFERENCES customers(customer_id),
    rating         INTEGER NOT NULL CHECK (rating BETWEEN 1 AND 5),
    comment        TEXT,
    review_date    TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    helpful_votes  INTEGER DEFAULT 0,
    UNIQUE (product_id, customer_id)
);

🎁 promotions

Discount codes and promotional offers

CREATE TABLE promotions (
    promo_id       SERIAL PRIMARY KEY,
    promo_code     VARCHAR(50) UNIQUE NOT NULL,
    discount_pct   DECIMAL(5,2) CHECK (discount_pct > 0 AND discount_pct <= 100),
    start_date     DATE NOT NULL,
    end_date       DATE NOT NULL,
    min_order      DECIMAL(10,2) DEFAULT 0.00,
    active         BOOLEAN DEFAULT TRUE,
    CHECK (end_date > start_date)
);

📜 audit_log

Track all data changes for auditing

CREATE TABLE audit_log (
    log_id         SERIAL PRIMARY KEY,
    table_name     VARCHAR(100) NOT NULL,
    operation      VARCHAR(10) NOT NULL CHECK (operation IN ('INSERT', 'UPDATE', 'DELETE')),
    old_data       JSONB,
    new_data       JSONB,
    changed_by     VARCHAR(100) DEFAULT CURRENT_USER,
    changed_at     TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

Quick Reference

Table Records Key Features Demonstrates
departments 5 SERIAL, VARCHAR, DECIMAL, TIMESTAMP Basic table, default values
employees 8 Self-reference, CHECK, JSONB Foreign keys, constraints, JSON
skills 8 CHECK constraint (ENUM-like) Data validation
employee_skills 12 Composite PK, ON DELETE CASCADE Many-to-many, junction table
customers 5 JSONB address, UNIQUE email JSON storage, uniqueness
categories 8 Self-referencing hierarchy Recursive relationships
suppliers 4 DEFAULT country Basic reference table
products 8 TEXT[] arrays, JSONB specs Arrays, JSON, multiple FKs
orders 7 Status CHECK, nullable shipped_date Status tracking, timestamps
order_items 11 Composite PK, ON DELETE CASCADE Order line items
reviews 6 UNIQUE constraint (product, customer) Rating system, composite unique
promotions 4 Date range CHECK Business rules, date constraints
audit_log 0 JSONB old/new data Audit trail, triggers

Recommended Indexes

-- Performance indexes for common queries
CREATE INDEX idx_employees_dept ON employees(dept_id);
CREATE INDEX idx_employees_manager ON employees(manager_id);
CREATE INDEX idx_employees_status ON employees(status);
CREATE INDEX idx_employees_metadata ON employees USING GIN(metadata);

CREATE INDEX idx_products_category ON products(category_id);
CREATE INDEX idx_products_supplier ON products(supplier_id);
CREATE INDEX idx_products_tags ON products USING GIN(tags);
CREATE INDEX idx_products_specs ON products USING GIN(specs);
CREATE INDEX idx_products_name_search ON products USING GIN(to_tsvector('english', product_name));

CREATE INDEX idx_orders_customer ON orders(customer_id);
CREATE INDEX idx_orders_employee ON orders(employee_id);
CREATE INDEX idx_orders_status ON orders(status);
CREATE INDEX idx_orders_date ON orders(order_date DESC);

CREATE INDEX idx_reviews_product ON reviews(product_id);
CREATE INDEX idx_reviews_rating ON reviews(rating);

CREATE INDEX idx_customers_email ON customers(email);
CREATE INDEX idx_customers_address ON customers USING GIN(address);
