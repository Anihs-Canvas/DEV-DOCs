4.5 COMPARISON EXAMPLE PATTERN (DO vs DON'T)
################################################################################

<div class="example-block example-comparison">
    <div class="example-header">
        <div class="example-title">
            <span class="example-number">Example 4</span>
            <span class="example-type-badge comparison">COMPARISON</span>
            <h5>count vs for_each - Choosing the Right Meta-Argument</h5>
        </div>
    </div>
    
    <p class="example-description">
        Understanding when to use count vs for_each is critical for proper 
        resource management and state stability.
    </p>
    
    <!-- Side by Side Comparison -->
    <div class="comparison-container">
        <div class="comparison-side dont">
            <div class="comparison-header">
                <span class="comparison-icon">❌</span>
                <span class="comparison-label">AVOID: Using count with lists</span>
            </div>
            <pre class="code-block terraform"><code># ❌ PROBLEM: count uses index-based addressing
variable "subnet_names" {
  default = ["web", "app", "db"]
}

resource "aws_subnet" "example" {
  count = length(var.subnet_names)
  
  vpc_id     = aws_vpc.main.id
  cidr_block = "10.0.${count.index}.0/24"
  
  tags = {
    Name = var.subnet_names[count.index]
  }
}

# Resources are addressed as:
# aws_subnet.example[0]  → "web"
# aws_subnet.example[1]  → "app"
# aws_subnet.example[2]  → "db"

# ⚠️ If you remove "app", index shifts!
# aws_subnet.example[1] → "db" (RECREATED!)
# aws_subnet.example[2] → DESTROYED!</code></pre>
        </div>
        
        <div class="comparison-side do">
            <div class="comparison-header">
                <span class="comparison-icon">✅</span>
                <span class="comparison-label">BETTER: Using for_each with sets/maps</span>
            </div>
            <pre class="code-block terraform"><code># ✅ SOLUTION: for_each uses key-based addressing
variable "subnet_names" {
  default = ["web", "app", "db"]
}

resource "aws_subnet" "example" {
  for_each = toset(var.subnet_names)
  
  vpc_id     = aws_vpc.main.id
  cidr_block = "10.0.${index(var.subnet_names, each.key)}.0/24"
  
  tags = {
    Name = each.key
  }
}

# Resources are addressed as:
# aws_subnet.example["web"]
# aws_subnet.example["app"]
# aws_subnet.example["db"]

# ✅ If you remove "app", only "app" is destroyed!
# Other resources remain untouched!</code></pre>
        </div>
    </div>
    
    <!-- Comparison Table -->
    <div class="comparison-table-container">
        <table class="comparison-table">
            <thead>
                <tr>
                    <th>Feature</th>
                    <th>count</th>
                    <th>for_each</th>
                </tr>
            </thead>
            <tbody>
                <tr>
                    <td>Addressing</td>
                    <td>Index-based <code>[0], [1], [2]</code></td>
                    <td>Key-based <code>["name"]</code></td>
                </tr>
                <tr>
                    <td>Best for</td>
                    <td>Identical resources, conditional creation</td>
                    <td>Distinct resources, dynamic configurations</td>
                </tr>
                <tr>
                    <td>On removal</td>
                    <td>⚠️ May recreate unrelated resources</td>
                    <td>✅ Only removes specific resource</td>
                </tr>
                <tr>
                    <td>Input type</td>
                    <td>Number</td>
                    <td>Set or Map</td>
                </tr>
                <tr class="highlight-exam">
                    <td>Exam Focus</td>
                    <td colspan="2">⚠️ Understanding the difference is EXAM CRITICAL</td>
                </tr>
            </tbody>
        </table>
    </div>

    <div class="scenario-box">

            </p>
            <ul>
                <li>VPC with public and private subnets</li>
                <li>EC2 instances in private subnet</li>
                <li>RDS MySQL in private subnet (depends on EC2 security group)</li>
                <li>Application Load Balancer in public subnet</li>
                <li>Zero-downtime deployments required</li>
            </ul>
        </div>
    </div>
    
    <div class="solution-section">
        <div class="solution-header">
            <span>🎯 Solution</span>
        </div>
        
        <pre class="code-block terraform"><code># Dependency chain visualization:
#
# ┌───────┐    ┌──────────┐    ┌─────────────────┐
# │  VPC  │───▶│ Subnets  │───▶│ Security Groups │
# └───────┘    └──────────┘    └────────┬────────┘
#                                       │
#              ┌────────────────────────┼─────────────────────┐
#              ▼                        ▼                     ▼
#         ┌─────────┐             ┌──────────┐          ┌─────────┐
#         │   EC2   │────────────▶│   RDS    │          │   ALB   │
#         │ (implicit)            │(explicit │          │         │
#         │  dep on SG)           │depends_on)          └─────────┘
#         └─────────┘             └──────────┘

# VPC (no dependencies)
resource "aws_vpc" "main" {
  cidr_block = "10.0.0.0/16"
  
  tags = { Name = "production-vpc" }
}

# Subnets (implicit dependency on VPC via vpc_id)
resource "aws_subnet" "private" {
  vpc_id     = aws_vpc.main.id  # Implicit dependency
  cidr_block = "10.0.1.0/24"
  
  tags = { Name = "private-subnet" }
}

# Security Group for EC2
resource "aws_security_group" "ec2" {
  vpc_id = aws_vpc.main.id
  
  ingress {
    from_port   = 8080
    to_port     = 8080
    protocol    = "tcp"
    cidr_blocks = ["10.0.0.0/16"]
  }
}

# Security Group for RDS
resource "aws_security_group" "rds" {
  vpc_id = aws_vpc.main.id
  
  ingress {
    from_port       = 3306
    to_port         = 3306
    protocol        = "tcp"
    security_groups = [aws_security_group.ec2.id]  # Allow from EC2 SG
  }
}

# EC2 Instance (implicit deps: subnet, security group)
resource "aws_instance" "app" {
  ami                    = data.aws_ami.amazon_linux.id
  instance_type          = "t3.medium"
  subnet_id              = aws_subnet.private.id
  vpc_security_group_ids = [aws_security_group.ec2.id]
  
  lifecycle {
    create_before_destroy = true
  }
  
  tags = { Name = "app-server" }
}

# RDS with EXPLICIT depends_on
resource "aws_db_instance" "mysql" {
  identifier           = "production-db"
  engine               = "mysql"
  instance_class       = "db.t3.medium"
  allocated_storage    = 20
  db_subnet_group_name = aws_db_subnet_group.main.name
  vpc_security_group_ids = [aws_security_group.rds.id]
  
  # Explicit dependency - ensures EC2 app is ready first
  # (even though no direct attribute reference exists)
  depends_on = [aws_instance.app]
  
  lifecycle {
    prevent_destroy = true  # Protect database from accidental deletion
  }
}</code></pre>
    </div>
    
    <!-- Exam Question Box -->
    <div class="exam-question-box">
        <div class="exam-question-header">
            <span class="exam-icon">📝</span>
            <strong>EXAM QUESTION</strong>
        </div>
        <div class="exam-question-content">
            <p><strong>Q:</strong> In the configuration above, why is <code>depends_on</code> 
            used for the RDS instance even though there's no direct attribute reference 
            to the EC2 instance?</p>
            
            <div class="exam-answer collapsed">
                <button onclick="toggleAnswer(this)">Show Answer ▼</button>
                <div class="answer-content">
                    <p><strong>A:</strong> The <code>depends_on</code> is used to handle 
                    <em>hidden dependencies</em> that Terraform cannot automatically detect. 
                    In this case, the application on the EC2 instance needs to be running 
                    before the database is created (perhaps for schema initialization or 
                    connection testing). Since there's no attribute reference from RDS 
                    to EC2, Terraform wouldn't know about this dependency without 
                    explicit declaration.</p>
                </div>
            </div>
        </div>

// THEME TOGGLE (Light/Dark Mode)
// ═══════════════════════════════════════════════════════════════════════════

function initTheme() {
    const savedTheme = localStorage.getItem('terraform-docs-theme') || 
                       (window.matchMedia('(prefers-color-scheme: dark)').matches ? 'dark' : 'light');
    setTheme(savedTheme);
}

function setTheme(theme) {
    document.documentElement.setAttribute('data-theme', theme);
    localStorage.setItem('terraform-docs-theme', theme);
    updateThemeIcon(theme);
}

function toggleTheme() {
    const current = document.documentElement.getAttribute('data-theme');
    const newTheme = current === 'dark' ? 'light' : 'dark';
    setTheme(newTheme);
}

function updateThemeIcon(theme) {
    const icon = document.getElementById('theme-icon');
    icon.textContent = theme === 'dark' ? '☀️' : '🌙';
}


// ═══════════════════════════════════════════════════════════════════════════
// TABLE OF CONTENTS NAVIGATION
// ═══════════════════════════════════════════════════════════════════════════

function toggleTOC() {
    const sidebar = document.getElementById('toc-sidebar');
    const main = document.getElementById('main-content');
    sidebar.classList.toggle('collapsed');
    main.classList.toggle('expanded');
}

function toggleSubTOC(element) {
    const parent = element.closest('.toc-item');
    const subList = parent.querySelector('.sub-toc');
    const icon = element.querySelector('.toggle-icon');
    
    if (subList) {
        subList.classList.toggle('collapsed');
        icon.textContent = subList.classList.contains('collapsed') ? '▶' : '▼';
    }
}

function scrollToSection(sectionId) {
    const section = document.getElementById(sectionId);
    if (section) {
        section.scrollIntoView({ behavior: 'smooth', block: 'start' });
        updateActiveTOCItem(sectionId);
        history.pushState(null, '', `#${sectionId}`);
    }
}

function updateActiveTOCItem(sectionId) {
    document.querySelectorAll('.toc-link').forEach(link => {
        link.classList.remove('active');
    });
    
    const activeLink = document.querySelector(`.toc-link[href="#${sectionId}"]`);
    if (activeLink) {
        activeLink.classList.add('active');
        expandParentTOCItems(activeLink);
    }
}

function expandParentTOCItems(element) {
    let parent = element.closest('.toc-item');
    while (parent) {
        const subToc = parent.querySelector('.sub-toc');
        if (subToc && subToc.classList.contains('collapsed')) {
            subToc.classList.remove('collapsed');
            const icon = parent.querySelector('.toggle-icon');
            if (icon) icon.textContent = '▼';
        }
        parent = parent.parentElement.closest('.toc-item');
    }
}


// ═══════════════════════════════════════════════════════════════════════════
// SEARCH FUNCTIONALITY
// ═══════════════════════════════════════════════════════════════════════════

const searchIndex = [];

function initSearch() {
    document.querySelectorAll('h2, h3, h4, h5').forEach(heading => {
        searchIndex.push({
            type: 'heading',
            level: parseInt(heading.tagName[1]),
            text: heading.textContent.toLowerCase(),
            id: heading.id || heading.closest('section')?.id,
            element: heading
        });
    });
    
    document.querySelectorAll('.example-block').forEach(example => {
        const title = example.querySelector('h5')?.textContent || '';
        searchIndex.push({
            type: 'example',
            text: title.toLowerCase(),
            id: example.id,
            element: example
        });
    });
}

function search(query) {
    if (query.length < 2) {
        hideSearchResults();
        return;
    }
    
    const results = searchIndex.filter(item => 
        item.text.includes(query.toLowerCase())
    ).slice(0, 10);
    
    displaySearchResults(results, query);
}

function displaySearchResults(results, query) {
    const container = document.getElementById('search-results');
    container.innerHTML = '';
    
    if (results.length === 0) {
        container.innerHTML = '<div class="no-results">No results found</div>';
    } else {
        results.forEach(result => {
            const item = document.createElement('div');
            item.className = 'search-result-item';
            item.innerHTML = `
                <span class="result-type ${result.type}">${result.type}</span>
                <span class="result-text">${highlightMatch(result.text, query)}</span>
            `;
            item.onclick = () => scrollToSection(result.id);
            container.appendChild(item);
        });
    }
    
    container.classList.remove('hidden');
}

function highlightMatch(text, query) {
    const regex = new RegExp(`(${escapeRegex(query)})`, 'gi');
    return text.replace(regex, '<mark>$1</mark>');
}

function escapeRegex(string) {
    return string.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
}

function hideSearchResults() {
    document.getElementById('search-results').classList.add('hidden');
}

// ═══════════════════════════════════════════════════════════════════════════
// TABLE FILTERING & SORTING
// ═══════════════════════════════════════════════════════════════════════════

function filterTable(input) {
    const query = input.value.toLowerCase();
    const table = input.closest('.table-container').querySelector('table');
    const rows = table.querySelectorAll('tbody tr');
    
    rows.forEach(row => {
        const text = row.textContent.toLowerCase();
        row.style.display = text.includes(query) ? '' : 'none';
    });
}

function filterByCategory(select) {
    const category = select.value;
    const table = select.closest('.table-container').querySelector('table');
    const rows = table.querySelectorAll('tbody tr');
    
    rows.forEach(row => {
        if (!category || row.dataset.category === category) {
            row.style.display = '';
        } else {
            row.style.display = 'none';
        }
    });
}

function sortTable(header, table) {
    const sortKey = header.dataset.sort;
    const tbody = table.querySelector('tbody');
    const rows = Array.from(tbody.querySelectorAll('tr'));
    const isAsc = header.classList.contains('sort-asc');
    
    table.querySelectorAll('th').forEach(th => {
        th.classList.remove('sort-asc', 'sort-desc');
    });
    
    rows.sort((a, b) => {
        const aVal = a.querySelector(`td:nth-child(${getColIndex(header)})`).textContent;
        const bVal = b.querySelector(`td:nth-child(${getColIndex(header)})`).textContent;
        return isAsc ? bVal.localeCompare(aVal) : aVal.localeCompare(bVal);
    });
    
    header.classList.add(isAsc ? 'sort-desc' : 'sort-asc');
    rows.forEach(row => tbody.appendChild(row));
}

function getColIndex(th) {
    return Array.from(th.parentElement.children).indexOf(th) + 1;
}

            });
        });
    });
}

function initScrollSpy() {
    const sections = document.querySelectorAll('section[id]');
    const observerOptions = { rootMargin: '-20% 0px -80% 0px' };
    
    const observer = new IntersectionObserver((entries) => {
        entries.forEach(entry => {
            if (entry.isIntersecting) {
                updateActiveTOCItem(entry.target.id);
            }
        });
    }, observerOptions);
    
    sections.forEach(section => observer.observe(section));
}

function initBackToTop() {
    const btn = document.getElementById('back-to-top');
    
    window.addEventListener('scroll', () => {
        btn.classList.toggle('visible', window.scrollY > 500);
    });
    
    btn.addEventListener('click', () => {
        window.scrollTo({ top: 0, behavior: 'smooth' });
    });
}

function initKeyboardShortcuts() {
    document.addEventListener('keydown', (e) => {
        if ((e.ctrlKey || e.metaKey) && e.key === 'k') {
            e.preventDefault();
            document.getElementById('search-input').focus();
        }
        if (e.key === 'Escape') {
            hideSearchResults();
            document.getElementById('search-input').blur();
        }
        if ((e.ctrlKey || e.metaKey) && e.key === 'b') {
            e.preventDefault();
            toggleTOC();
        }
        if ((e.ctrlKey || e.metaKey) && e.key === 'd') {
            e.preventDefault();
            toggleTheme();
        }
    });
}

// PROGRESS TRACKING (Local Storage)
// ═══════════════════════════════════════════════════════════════════════════

const progressKey = 'terraform-docs-progress';

function getProgress() {
    return JSON.parse(localStorage.getItem(progressKey) || '{}');
}

function markChapterComplete(chapterId) {
    const progress = getProgress();
    progress[chapterId] = { completed: true, timestamp: new Date().toISOString() };
    localStorage.setItem(progressKey, JSON.stringify(progress));
    updateProgressUI();
}

function updateProgressUI() {
    const progress = getProgress();
    const totalChapters = 55;
    const completedCount = Object.values(progress).filter(p => p.completed).length;
    const percentage = Math.round((completedCount / totalChapters) * 100);
    
    const progressBar = document.querySelector('.progress-fill');
    if (progressBar) progressBar.style.width = `${percentage}%`;
    
    const progressText = document.querySelector('.progress-percentage');
    if (progressText) progressText.textContent = `${percentage}% Complete`;
    
    Object.keys(progress).forEach(chapterId => {
        if (progress[chapterId].completed) {
            const tocItem = document.querySelector(`.toc-link[href="#${chapterId}"]`);
            if (tocItem) tocItem.classList.add('completed');
        }
    });
}

################################################################################
7.4 CODE EXAMPLE QUALITY STANDARDS
################################################################################

All code examples MUST follow these standards:

✅ DO:
─────────────────────────
• Use realistic, production-like names (not "foo", "bar", "test")
• Include ALL required arguments
• Add comments explaining non-obvious parts
• Show expected output/results
• Use consistent indentation (2 spaces for HCL)
• Include variable declarations when used
• Tag resources with meaningful metadata

❌ DON'T:
─────────────────────────
• Use placeholder values without explanation
• Skip required arguments "for brevity"
• Use outdated syntax or deprecated features
• Mix different provider versions
• Hardcode sensitive values
• Use single-letter variable names


GOOD EXAMPLE FORMAT:
─────────────────────────
# Description of what this example demonstrates
# and when you would use this pattern

variable "environment" {
  description = "Deployment environment (dev/staging/prod)"
  type        = string
  default     = "dev"
  
  validation {
    condition     = contains(["dev", "staging", "prod"], var.environment)
    error_message = "Environment must be dev, staging, or prod."
  }
}

resource "aws_instance" "web_server" {
  # Using the latest Amazon Linux 2 AMI from data source
  ami           = data.aws_ami.amazon_linux.id
  instance_type = var.environment == "prod" ? "t3.medium" : "t3.micro"
  
  # Security group allowing HTTPS traffic
  vpc_security_group_ids = [aws_security_group.web.id]
  
  tags = {
    Name        = "web-server-${var.environment}"
    Environment = var.environment
    ManagedBy   = "Terraform"
    Project     = "terraform-docs"
  }
}
